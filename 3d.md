# 3D Earth Globe Dashboard Feature

This implementation uses React Three Fiber and Drei to render an interactive Earth globe in the dashboard. Clicking a country reveals market statistics in a side panel.

## Installation

```bash
npm install three @react-three/fiber @react-three/drei topojson-client
```

## Example Component

```jsx
import { Canvas, useFrame } from '@react-three/fiber'
import { OrbitControls } from '@react-three/drei'
import { useMemo, useState, useRef } from 'react'
import * as THREE from 'three'
import { feature } from 'topojson-client'
import worldData from './data/world-110m.json'

function CountryMesh({ shape, onSelect }) {
  const ref = useRef()
  const color = useMemo(() => new THREE.Color('#4682b4'), [])
  const hoverColor = new THREE.Color('#ffcc00')

  return (
    <mesh
      ref={ref}
      geometry={shape}
      onPointerOver={() => (ref.current.material.color = hoverColor)}
      onPointerOut={() => (ref.current.material.color = color)}
      onClick={onSelect}
    >
      <meshStandardMaterial attach="material" color={color} roughness={1} metalness={0}
      />
    </mesh>
  )
}

function EarthGlobe({ countries, onCountryClick }) {
  return (
    <group rotation={[0, 0, 0]}>
      <mesh>
        <sphereGeometry args={[2.5, 64, 64]} />
        <meshStandardMaterial color="#0a2f52" roughness={0.8} metalness={0.1} />
      </mesh>

      {countries.map((country) => (
        <CountryMesh
          key={country.id}
          shape={country.shape}
          onSelect={() => onCountryClick(country.properties)}
        />
      ))}
    </group>
  )
}

export default function GlobeDashboard() {
  const [selectedCountry, setSelectedCountry] = useState(null)

  const countries = useMemo(() => {
    const geo = feature(worldData, worldData.objects.countries)
    return geo.features.map((feature) => {
      const shapes = new THREE.ShapeGeometry(
        new THREE.Shape(feature.geometry.coordinates[0].map(([lng, lat]) => {
          const phi = (90 - lat) * (Math.PI / 180)
          const theta = (lng + 180) * (Math.PI / 180)
          const radius = 2.51
          return new THREE.Vector3(
            radius * Math.sin(phi) * Math.cos(theta),
            radius * Math.cos(phi),
            radius * Math.sin(phi) * Math.sin(theta)
          )
        }))
      )
      return { id: feature.id, shape: shapes, properties: feature.properties }
    })
  }, [])

  return (
    <div style={{ display: 'flex', height: '100vh' }}>
      <div style={{ flex: 1 }}>
        <Canvas camera={{ position: [0, 0, 8], fov: 50 }}>
          <ambientLight intensity={0.6} />
          <directionalLight position={[5, 5, 5]} intensity={1} />
          <OrbitControls enableZoom={true} />
          <EarthGlobe countries={countries} onCountryClick={setSelectedCountry} />
        </Canvas>
      </div>

      <aside style={{ width: '320px', padding: '20px', background: '#101828', color: '#fff' }}>
        <h2>Market Analysis</h2>
        {selectedCountry ? (
          <div>
            <h3>{selectedCountry.name || 'Unknown Country'}</h3>
            <p>Market status: {selectedCountry.marketStatus || 'No data available'}</p>
            <p>GDP: {selectedCountry.gdp || 'N/A'}</p>
            <p>Trade volume: {selectedCountry.tradeVolume || 'N/A'}</p>
          </div>
        ) : (
          <p>Click a country on the globe to see market statistics.</p>
        )}
      </aside>
    </div>
  )
}
```

## Notes

- Use a proper GeoJSON/TopoJSON data source for country shapes.
- Extend `selectedCountry` state with real market data from your analytics backend.
- Adjust styling and camera controls for dashboard layout.
